Note: You have to complete Print_pattern. No need to take any input.

      const Print_pattern = (N) => 
      {
        let add = "";
        for (let i = 1; i <= N; i++) {
          for (let j = 0; j < i; j++) {
            add += "*";
          }
           add += "\n";
        }
        console.log(add);
      };
